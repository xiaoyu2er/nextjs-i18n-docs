`draftMode` 関数を使用すると、[サーバ゙ーコンポーネント (Server Component)](/docs/app/building-your-application/rendering/server-components) 内で[デラフトモード (Draft Mode)](/docs/app/building-your-application/configuring/draft-mode) を検出できます。

```jsx filename="app/page.js"
import { draftMode } from 'next/headers'

export default function Page() {
  const { isEnabled } = draftMode()
  return (
    <main>
      <h1>My Blog Post</h1>
      <p>Draft Mode is currently {isEnabled ? 'Enabled' : 'Disabled'}</p>
    </main>
  )
}
```
