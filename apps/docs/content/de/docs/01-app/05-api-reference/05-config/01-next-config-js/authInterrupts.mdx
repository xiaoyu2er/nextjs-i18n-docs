Die Konfigurationsoption `authInterrupts` ermöglicht die Verwendung der APIs [`forbidden`](/docs/app/api-reference/functions/forbidden) und [`unauthorized`](/docs/app/api-reference/functions/unauthorized) in Ihrer Anwendung. Da diese Funktionen experimentell sind, müssen Sie die Option `authInterrupts` in Ihrer `next.config.js`-Datei aktivieren, um sie nutzen zu können:

```ts filename="next.config.ts" switcher
import type { NextConfig } from 'next'

const nextConfig: NextConfig = {
  experimental: {
    authInterrupts: true,
  },
}

export default nextConfig
```

```js filename="next.config.js" switcher
module.exports = {
  experimental: {
    authInterrupts: true,
  },
}
```
