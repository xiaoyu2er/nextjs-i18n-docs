`draftMode` 函式可讓您在[伺服器元件 (Server Component)](/docs/app/building-your-application/rendering/server-components) 中偵測[草稿模式 (Draft Mode)](/docs/app/building-your-application/configuring/draft-mode)。

```jsx filename="app/page.js"
import { draftMode } from 'next/headers'

export default function Page() {
  const { isEnabled } = draftMode()
  return (
    <main>
      <h1>我的部落格文章</h1>
      <p>草稿模式目前為 {isEnabled ? '啟用' : '停用'}</p>
    </main>
  )
}
```
